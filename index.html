        function runAssessmentSearch() {

            const input = document.getElementById("assessmentSearchInput");

            const filter = input.value.toUpperCase();

            const table = document.getElementById("assessmentTable");

            const tr = table.getElementsByTagName("tr");


            for (let i = 1; i < tr.length; i++) {

                // Get all text content from the row's cells

                const tdCat = tr[i].getElementsByTagName("td")[0];

                const tdType = tr[i].getElementsByTagName("td")[1];

                const tdCode = tr[i].getElementsByTagName("td")[2];

                

                if (tdCat && tdType) {

                    const textValue = (tdCat.textContent || tdCat.innerText) + 

                                     (tdType.textContent || tdType.innerText) + 

                                     (tdCode.textContent || tdCode.innerText);

                    

                    tr[i].style.display = textValue.toUpperCase().indexOf(filter) > -1 ? "" : "none";

                }

            }

        }


        function copyToClipboard(text, btn) {

            navigator.clipboard.writeText(text).then(() => {

                const originalText = btn.innerText;

                const originalClasses = btn.className;

                

                // Change button appearance to green "Copied!" state

                btn.innerText = "COPIED!";

                btn.classList.add("copy-feedback");

                

                // Reset after 1.5 seconds

                setTimeout(() => {

                    btn.innerText = originalText;

                    btn.classList.remove("copy-feedback");

                }, 1500);

            });

        }
